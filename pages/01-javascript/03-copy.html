👶 개요

웹 페이지의 코드를 손쉽게 복사하는 방법이 궁금해서 찾아봤다.
드래그만 하면 자동으로 복사되는 기능이 있을까?ㅎㅎ
2009년 개발자: 됩니다.
클릭만 해도 복사되는 기능도 있을까?ㅎㅎ
2015년 개발자: 되용.

사람들 생각하는 건 다 비슷하다는 것을 또 한번 느낀 시간이었다.

🍞 클립보드 복사 코드 

                
                  // Copy ClickEvent
                  let parentElement = document.querySelector('.div');

                  function copyData (elem) {
                    let dummy = document.createElement('textarea');
                    document.body.appendChild(dummy);
                    dummy.value = elem.target.innerText;
                    dummy.select();
                    document.execCommand('copy');
                    document.body.removeChild(dummy);
                    alert(dummy.value + '복사됨');
                  }

                  function copyHandler (e) {
                    copyData(e);
                  }

                  parentElement.addEventListener('click', copyHandler);

🏀 createElement()

HTML문서에서, 지정한 tagName의 HTML 요소를 만들어 반환하는 메소드

syntax
let element = document.createElement(tagName[, options]);
createElement를 하는 이유
처음에, 변수만 필요하면 될 줄 알았는데, execCommand메소드가 변수의 데이터를 인식하지 못했다. 대상을 명시적으로 표현하는 장치가 필요하다는 말인데, 이 때 필요한 게 select 메소드다. execCommand메소드를 선언하기 이전에 select메소드를 선언하면 된다.
하지만 인생이 이렇게 잘 풀릴리가 없다

select메소드는 textarea나 input(type=text 또는 type=url)에서만 제 역할을 한다. input말고 다른 요소의 텍스트 가져다 쓰고싶다면, createElement메소드를 이용하면 된다.
명령 순서
createElement메소드로 태그 만들고
append로 태그 추가하고
input.value에 텍스트를 저장하고
select를 input에 선언하고
removeChild로 태그 삭제한다.

🏀 select()

textarea나 input(text, url 등 텍스트를 담아내는 타입)의 모든 텍스트를 선택하는 메소드

syntax
InputElement.select();
크로스 브라우징
InputElement.setSelectionRange();
괄호 안에는 (시작점, 끝점) 입력한다.
예시)
this.setSelectionRange(0, this.value.length);
적용 대상
textarea - 여백 유지
input - 여백 취합

🏀 execCommand()

선택된 텍스트를 편집하는 메소드
문법은 다음과 같다.

document.execCommand(aCommandName, aShowDefaultUI, aValueArgument)

👶 input에서 복사

HTML의 input요소의 텍스트를 가져온다.
IE 10+, Chrome 43+, Opera 29+, Firefox 가능 

                
                  function copy_to_clipboard() {
                    const dummy = document.querySelector("input");
                    dummy.select();
                    document.execCommand("Copy");
                  }

👷 구식 자동/수동 혼합

익스플로러가 아닐 경우 prompt를 띄워 수동으로 복사 하도록 함 

                
                  function is_ie() {
                    if(navigator.userAgent.toLowerCase().indexOf("chrome") != -1) return false;
                    if(navigator.userAgent.toLowerCase().indexOf("msie") != -1) return true;
                    if(navigator.userAgent.toLowerCase().indexOf("windows nt") != -1) return true;
                    return false;
                  }

                  function copy_to_clipboard(str) {
                    if( is_ie() ) {
                      window.clipboardData.setData("Text", str);
                      alert("복사되었습니다.");
                      return;
                    }
                    prompt("Ctrl+C를 눌러 복사하세요.", str);
                  }
navigator
브라우저의 종류와 버전 등 웹브라우저 전반에 대한 정보를 제공하는 객체
navigator.userAgent
브라우저가 User-Agent HTTP 헤더에 넣어 전송하는 문자열로 appName과 appVersion의 모든 정보를 포함하며 더 자세한 정보를 추가로 담고 있다.
