😋 개요

구현하는 건 어렵지 않다.
그런데 왜 정리하냐면 내 마음이다.

😵 구현

구성

  • 이벤트 리스너(mouseup, mousedown)
  • 변수(좌표저장, 플래그, 방향)

주의사항

  • 시작점과 끝점의 좌표
  • 변수의 스코프
  • 플래그의 조건과 동작문의 위치
예시 코드
            
              var isDragging = false;
              var lastX;
              var lastY;
              var 방향;

              document.addEventListener('mouseup', (e) => {
                var x차이 = e.offsetX - lastX;
                var y차이 = e.offsetY - lastY;

                if (x차이 === 0 && y차이 === 0) { // 클릭일 때,
                  isDragging = false;
                  return;
                } else { // 드래그일 때,
                  isDragging = true;
                }

                if (isDragging) {
                  if (y차이 > 0 && Math.abs(x차이) / Math.abs(y차이) < 1) {
                    방향 = "아래";
                  } else if (y차이 < 0 && Math.abs(x차이) / Math.abs(y차이) < 1) {
                    방향 = "위";
                  } else if (x차이 < 0 && Math.abs(x차이) / Math.abs(y차이) > 1) {
                    방향 = "왼쪽";
                  } else if (x차이 > 0 && Math.abs(x차이) / Math.abs(y차이) > 1) {
                    방향 = "오른쪽";
                  }
                }
                console.log(방향);
              });

              document.addEventListener('mousedown', (e) => {
                [lastX, lastY] = [e.offsetX, e.offsetY];
              });
